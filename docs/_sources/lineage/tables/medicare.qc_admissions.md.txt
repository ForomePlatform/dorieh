Aggregate data for admissions, beneficiaries and enrollments


Transformed from [medicare.qc_adm_union](medicare.qc_adm_union.md)


<details>

```sql

CREATE materialized view  medicare.qc_admissions AS
SELECT
    year,
	state,
	zip,
	reason,
	MAX(state_iso) AS state_iso,
	COUNT(*) AS NumRecords,
	(#(hll_add_agg(bene)))::BIGINT AS NumDistinctBeneficaries,
	hll_add_agg(bene) AS bene_hll
FROM medicare.qc_adm_union

GROUP BY year,state,zip,reason;

COMMENT ON materialized view medicare.qc_admissions IS 'CREATED BY Dorieh: {"version": "0.2.1", "url": "https://github.com/NSAPH-Data-Platform/dorieh", "commit": "0a20037a78bb2a1a32bffc845e5c68742da04afd"}. Created at 2024-08-09 17:28:01.297049';
```
